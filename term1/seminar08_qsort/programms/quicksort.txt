def Quicksort(A, p, r):
    """Sort array A in place.

    Initial call should be Quicksort(A, 0, len(A))
    """
    if p < r:
	q = Partition(A, p, r)
	Quicksort(A, p, q)
	Quicksort(A, q+1, r)



def Partition(A, p, r):
    x = A[p]
    i = p - 1
    j = r + 1
    while 1:

		j = j - 1
		while A[j] <= x:
			j = j - 1

		i = i + 1
		while A[i] >= x:
			j = j - 1

		if i < j:
			swap(A[i], A[j])
		else:
			return j
